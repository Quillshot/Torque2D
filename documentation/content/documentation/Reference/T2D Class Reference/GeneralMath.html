GeneralMath Description: General math functions. Use these whenever possible, as they'll run much faster than script equivalents.

Methods:

mAbs(float v)

Description: Returns the absolute value of the argument.

mAcos(float th)

Description: Returns the arc-cosine of th, which is in radians.

mAsin(float th)

Description: Returns the arc-sine of th, which is in radians.

mAtan(float rise, float run)

Description: Returns the slope in radians (the arc-tangent) of a line with the given rise and run.

mCeil(float v)

Description: Round v up to the nearest whole number.

mClamp(float number, float min, float max)

Description: Clamp a value between two other values.

mCos(float th)

Description: Returns the cosine of th, which is in radians.

mDegToRad(float degrees)

Description: Convert a measure in degrees to radians.

mFloatLength(float v, int numDecimals)

Description: Return a string containing v formatted with the specified number of decimal places.

mFloor(float v)

Description: Round v down to the nearest whole number.

mLog(float v)

Description: Returns the natural logarithm of the argument.

mPow(float b, float p)

Description: Returns the b raised to the pth power.

mRadToDeg(float radians)

Description: Converts a measure in radians to degrees.

mRound(float v)

Description: Rounds a number

mSin(float th)

Description: Returns the sine of th, which is in radians.

mSolveCubic(float a, float b, float c, float d)

Description: Solve a cubic equation of form a*x^3 + b*x^2 + c*x + d = 0.

mSolveQuadratic(float a, float b, float c)

Description: Solve a quadratic equation of form a*x^2 + b*x + c = 0.

mSolveQuartic(float a, float b, float c, float d, float e)

Description: Solve a quartic equation of form a*x^4 + b*x^3 + c*x^2 + d*x + e = 0.

mSqrt(float v)

Description: Returns the square root of the argument.

mTan(float th)

Description: Returns the tangent of th, which is in radians.
